CREATE TABLE account_info(
    login BIGINT,
    trade_mode INT,
    leverage INT,
    limit_orders INT,
    margin_so_mode INT,
    trade_allowed BOOLEAN,
    trade_expert BOOLEAN,
    margin_mode INT,
    currency_digits INT,
    fifo_close BOOLEAN,
    balance DOUBLE,
    credit DOUBLE,
    profit DOUBLE,
    equity DOUBLE,
    margin DOUBLE,
    margin_free DOUBLE,
    margin_level DOUBLE,
    margin_so_call DOUBLE,
    margin_so_so DOUBLE,
    margin_initial DOUBLE,
    margin_maintenance DOUBLE,
    assets DOUBLE,
    liabilities DOUBLE,
    commission_blocked DOUBLE,
    name VARCHAR(200),
    server VARCHAR(200),
    currency VARCHAR(100),
    company VARCHAR(200),
    creation_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    modification_time DATETIME ON UPDATE CURRENT_TIMESTAMP,
    _date VARCHAR(40),
);


CREATE TABLE positions_info(
    ticket BIGINT,
    time VARCHAR(100),
    type INT,
    magic INT,
    identifier BIGINT,
    reason INT,
    volume DOUBLE,
    price_open DOUBLE,
    sl DOUBLE,
    tp DOUBLE,
    price_current DOUBLE,
    swap DOUBLE,
    profit DOUBLE,
    symbol VARCHAR(200),
    comment TEXT,
    creation_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    modification_time DATETIME ON UPDATE CURRENT_TIMESTAMP,
    login BIGINT,
    _date VARCHAR(40)
);


CREATE TABLE trade_history(
    ticket BIGINT,
    time_setup VARCHAR(100),
    time_setup_msc BIGINT,
    time_done VARCHAR(100),
    time_done_msc BIGINT,
    type INT,
    type_filling INT,
    magic INT,
    position_id BIGINT,
    volume_initial DOUBLE,
    price_open DOUBLE,
    price_current DOUBLE,
    symbol VARCHAR(200),
    comment TEXT,
    external_id VARCHAR(200),
    creation_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    modification_time DATETIME ON UPDATE CURRENT_TIMESTAMP,
    login BIGINT,
    _date VARCHAR(40)
);
